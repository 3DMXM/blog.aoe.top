代码：

```js
(function($) {
    /**
     * 获取传递的get值
     * @param name
     * @returns {string|null}
     */
    $.getUrlParam = function(name) {
        let reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)");
        let r = window.location.search.substr(1).match(reg);
        if(r != null) return decodeURI(r[2]);
        return null;
    }
    /**
     * 程序休息
     * @param time
     * @returns {Promise<unknown>}
     */
    $.sleep = function (time) {
        return new Promise((resolve) => setTimeout(resolve, time));
    }
})(jQuery);
```
使用方法：
